data.naomited <- activity.data
cnt <- 0
for (i in 1:nrow(data.naomited)) {

    if (is.na(data.naomited$steps[i])) {
        data.naomited$steps[i] <- intervals.average[data.naomited$interval[i] ==

        cnt <- cnt + 1
    }
